    <script>
      document.querySelectorAll(".draggable").forEach((item) => {
        item.addEventListener("mousedown", mouseDownHandler);
      });

      function mouseDownHandler(e) {
        const svg = document.querySelector("svg");
        const rect = svg.getBoundingClientRect();
        const item = e.target.closest(".card-item");
        const offsetX =
          e.clientX - rect.left - parseFloat(item.getAttribute("x"));
        const offsetY =
          e.clientY - rect.top - parseFloat(item.getAttribute("y"));

        function mouseMoveHandler(e) {
          const newX = e.clientX - rect.left - offsetX;
          const newY = e.clientY - rect.top - offsetY;
          item.setAttribute("x", newX);
          item.setAttribute("y", newY);

          // 更新路径数据
          updateConnectionPath();
        }

        function mouseUpHandler() {
          document.removeEventListener("mousemove", mouseMoveHandler);
          document.removeEventListener("mouseup", mouseUpHandler);
        }

        document.addEventListener("mousemove", mouseMoveHandler);
        document.addEventListener("mouseup", mouseUpHandler);
      }

      function updateConnectionPath() {
        const path = document.getElementById("connectionPath");

        const studentRect = student.getBoundingClientRect();
        const courseRect = course.getBoundingClientRect();

        const studentX =
          parseFloat(student.getAttribute("x")) + studentRect.width;
        const studentY = parseFloat(student.getAttribute("y")) + 24 + 12;
        const courseX = parseFloat(course.getAttribute("x")) - 15 - 20;
        const courseY = parseFloat(course.getAttribute("y")) + 24 + 26 * 3 - 13;

        const d = `M ${studentX} ${studentY} L ${
          studentX + 20
        } ${studentY} L ${courseX} ${courseY} L ${courseX + 20} ${courseY}`;
        path.setAttribute("d", d);
      }

      function toggleCollapse(id) {
        const cardItem = document.getElementById(id);
        const contentDiv = cardItem.querySelector(".content");
        if (contentDiv.classList.contains("collapsed")) {
          contentDiv.classList.remove("collapsed");
          cardItem.querySelector("button").textContent = "-";
        } else {
          contentDiv.classList.add("collapsed");
          cardItem.querySelector("button").textContent = "+";
        }
      }
    </script>
